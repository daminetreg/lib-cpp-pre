137  ('6', 6)('7', 7)('8', 8)
138  ('9', 9)('a', 10)('b', 11)

142  highNibbleChar = tolower(highNibbleChar);
143  }
144 
145  if (!isdigit(lowNibbleChar)) {
146  lowNibbleChar = tolower(lowNibbleChar);
147  }
148

151  boost::uint8_t lowNibble = charToNibble.find(lowNibbleChar)->second;
